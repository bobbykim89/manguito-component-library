# MCL Input Radio

Included in `@bobbykim/mcl-forms`

## Installation

```sh
npm i @bobbykim/mcl-forms
```

## Usage

In your code

```js
<script setup lang="ts">
import { MclInputRadio } from '@bobbykim/mcl-forms'
import { ref } from 'vue'

const valueRef = ref<string>('')
const sampleData = {
  id: 'item-1',
  value: 'first-item',
}

const setValue = (e: Event, value: string | number) => {
  valueRef.value = value
}
</script>

<template>
  <mcl-input-switch :id="sampleData.id" :value="sampleData.value" :checked="valueRef === sampleData.value" @change="setValue"></mcl-input-switch>
</template>
```

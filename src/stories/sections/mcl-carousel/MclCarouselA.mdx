# MCL Carousel A

Included in `@bobbykim/mcl-carousel`

## Install

```zsh
npm i @bobbykim/mcl-carousel
```

## Usage

In your code

```js
<script setup lang="ts">
import { MclCollapseA } from '@bobbykim/mcl-carousel'
</script>

<template>
  <carousel-alpha>
    <!-- adding nested card-components -->
    <template #carousel="{ setRef, cards }">
      <card-component
        v-for="(card, i) in cards"
        :key="i"
        :ref="(el) => setRef(el)"
        class="mr-xs last:mr-0 h-full"
      ></card-component>
    </template>
  </carousel-alpha>
</template>
```

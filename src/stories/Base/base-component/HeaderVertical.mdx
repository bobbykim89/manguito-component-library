# Header Vertical

Included in `@bobbykim/manguito-theme`

## Usage

```js
<script lang="ts" setup>
import { HeaderVertical } from '@bobbykim/manguito-theme'

const headerRef = ref<InstanceType<typeof HeaderVertical>>()

const handleMenuClick = (e: Event) => {
  e.preventDefault()
  headerRef.value!.headerClose()
}
</script>

<template>
  <HeaderVertical ref="headerRef" v-bind="props">
    <template #content>
      <div class="flex flex-col flex-shrink-0 items-center self-center">
        <!-- content -->
    </template>
    <template #content-bottom>
      <!-- header right content -->
    </template>
    <template #mobile-content="{ headerClose }">
      <!-- mobile menu content -->
      <div>
        <a href="#" @click="handleMenuClick($event)">Mobile Menu 1</a>
        <a href="#" @click="handleMenuClick($event)">Mobile Menu 2</a>
        <a href="#" @click="handleMenuClick($event)">Mobile Menu 3</a>
        <div>
          <button class="btn btn-primary" @click="headerClose">
            Button A
          </button>
          <button class="btn btn-secondary" @click="headerClose">
            Button B
          </button>
        </div>
      </div>
    </template>
  </HeaderVertical>
</template>
```

## Slot

- `#content`: main content slot on top of the component
- `#content-bottom`: slot at the bottom of the component
- `#mobile-content`: slot inside expanded mobile menu

## Events

-`toggle-drawer` - (event: Event) - Emitted nav drawer button click

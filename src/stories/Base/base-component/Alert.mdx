# Alert

Included in `@bobbykim/manguito-theme`

## Usage

```js
<script lang="ts" setup>
import { Alert } from '@bobbykim/manguito-theme'

const alertRef = ref<InstanceType<typeof Alert>>()

// to handle through exposed function
const handleMenuClick = (e: Event) => {
  e.preventDefault()
  alertRef.value!.close()
}
</script>

<template>
  <Alert show ref="alertRef">
    <div>
      Lorem ipsum dolor sit amet consectetur adipisicing elit. Est tenetur impedit hic iure, consectetur cupiditate nesciunt ullam voluptatum veniam ipsam?
    </div>
  </Alert>
</template>
```
